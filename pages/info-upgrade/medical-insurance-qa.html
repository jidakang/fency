页面内容采集

页面内容提取

通过JavaScript提取当前页面的文本内容，作为问答的上下文信息：

extractPageContent.js
// 提取页面文本内容函数
function extractPageContent() {
  // 使用textContent属性提取页面可见文本
  const mainContent = document.querySelector('main, .content, #content, article, .article');
  let pageText = '';
  
  if (mainContent) {
    // 优先提取主内容区域
    pageText = mainContent.textContent;
  } else {
    // 备选：提取body内容但排除导航、页脚等通用元素
    const body = document.body;
    const elementsToExclude = 'nav, footer, header, .navigation, .menu, .footer';
    
    // 创建一个临时副本以便修改
    const bodyClone = body.cloneNode(true);
    const excludeElements = bodyClone.querySelectorAll(elementsToExclude);
    
    // 移除不需要的元素
    excludeElements.forEach(el => el.parentNode.removeChild(el));
    pageText = bodyClone.textContent;
  }
  
  // 清理文本（移除多余空格和分行）
  return pageText.replace(/\s+/g, ' ').trim();
}

DeepSeek API 交互

API调用

实现与已部署的DeepSeek API服务的集成，支持流式响应：

apiService.js
async function askQuestion(question, pageContent) {
  const apiEndpoint = '/api/deepseek/ask'; // 已部署的API端点
  
  try {
    // 显示加载状态
    showLoadingIndicator();
    
    const response = await fetch(apiEndpoint, {
      method: 'POST',
      headers: {
        'Content-Type': 'application/json'
      },
      body: JSON.stringify({
        question: question,
        context: pageContent,
        stream: true
      })
    });
    
    // 处理流式响应
    const reader = response.body.getReader();
    const decoder = new TextDecoder();
    let answer = '';
    
    while (true) {
      const { done, value } = await reader.read();
      if (done) break;
      
      // 解码收到的数据块
      const chunk = decoder.decode(value, { stream: true });
      answer += chunk;
      
      // 实时更新UI
      updateAnswerInUI(answer);
    }
    
    hideLoadingIndicator();
    return answer;
  } catch (error) {
    hideLoadingIndicator();
    return '抱歉，发生了错误，请稍后再试。';
  }
}
